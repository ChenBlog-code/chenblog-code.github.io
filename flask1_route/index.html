from flask import Flask

app =Flask(__name__)

@app.route('/test/')
def test():
    return "this is test route"

#pass one param  : 127.0.0.1:5001/arg/baichen/
@app.route('/arg/<name>/')
def arg(name):
    return "hello "+name

#pass many params
@app.route('/arg/<name>/<age>/')
@app.route('/arg/<name>_<age>/')
def args(name,age):
    return "Name: {} Age: {}".format(name,age)

if __name__ == '__main__':
    app.run(port=5001,debug=True)

url :

  • 127.0.0.1/test/
  • 127.0.0.1/arg/chen/
  • 127.0.0.1:5000/arg/chen/10/
  • 127.0.0.1:5000/arg/chen_10/

Param type:

#type of param
@app.route('/test/<name>/')
def test(name):
    print(type(name))    # it is string type
    return "testing type of param"

if __name__ == '__main__':
    app.run(port=5001,debug=True)

Default : name is string type

url: http://127.0.0.1:5001/test/chen/

Limit the arg type to be integer:

#type of param
@app.route('/test/<int:arg>/')  #Limit the arg type to be integer:
def test(arg):
    print(type(arg))    # it is int
    return "testing type of param"

if __name__ == '__main__':
    app.run(port=5001,debug=True)

url: http://127.0.0.1:5001/test/10/

Path: arg “a/b/c/”

#type of param: path is a string
@app.route('/test/<path:arg>/')
def test(arg):
    print(type(arg))
    print(arg)
    return "testing type of param"

if __name__ == '__main__':
    app.run(port=5001,debug=True)

url: http://127.0.0.1:5001/test/a/b/c/
